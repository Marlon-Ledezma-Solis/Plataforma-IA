Utilidades IQR: Funciones Base
import pandas as pd
import numpy as np

def tukey_bounds(series: pd.Series, k: float = 1.5):
 """Devuelve límites (low, high) por Tukey."""
 s = series.dropna()
 if len(s) < 6: # mínimo para estabilidad
 return np.nan, np.nan
 
 q1 = s.quantile(0.25)
 q3 = s.quantile(0.75)
 iqr = q3 - q1
 low = q1 - k * iqr
 high = q3 + k * iqr
 return low, high

def tukey_flag(series: pd.Series, k: float = 1.5):
 low, high = tukey_bounds(series, k=k)
 if np.isnan(low) or np.isnan(high):
 return pd.Series(False, index=series.index), low, high
 return (series < low) | (series > high), low, high

def detect_customer_outliers(orders: pd.DataFrame, k: float = 1.5):
 d = orders.copy()
 
 def per_customer(g):
 g = g.sort_values(["order_date","order_id"]).copy()
 
 flag_rev, rev_low, rev_high = tukey_flag(g["order_revenue"], k=k)
 flag_gap, gap_low, gap_high = tukey_flag(g["days_since_prev_purchase"], k=k)
 
 g["outlier_order_revenue"] = flag_rev
 g["rev_low"] = rev_low
 g["rev_high"] = rev_high
 g["outlier_gap_days"] = flag_gap
 g["gap_low"] = gap_low
 g["gap_high"] = gap_high
 g["any_customer_outlier"] = g["outlier_order_revenue"] | g["outlier_gap_days"]

def add_mom_yoy(df, value_cols, yoy_lag=12):
 out = df.copy().sort_values("month")
 for c in value_cols:
 out[f"{c}_mom_pct"] = out[c].pct_change() * 100
 out[f"{c}_yoy_pct"] = out[c].pct_change(yoy_lag) * 100
 return out

--- Diapositiva 67 ---
Factor Estacional: Concepto
Definición
Sm > 1: mes por encima del promedio
Sm < 1: mes por debajo del promedio
Interpretación
El factor estacional cuantifica cuánto se desvía cada mes calendario del comportamiento promedio. Es multiplicativo y no asume normalidad.
Por ejemplo, si diciembre tiene Sm = 1.5, significa que típicamente genera 50% más ingresos que el mes promedio.

def tukey_bounds(series, k=1.5):
 s = series.dropna()
 if len(s) < 8:
 return np.nan, np.nan
 q1, q3 = s.quantile(0.25), s.quantile(0.75)
 iqr = q3 - q1
 return q1 - k*iqr, q3 + k*iqr

def add_tukey_flag(df, change_col, k=1.5):
 out = df.copy()
 low, high = tukey_bounds(out[change_col], k=k)
 out[f"flag_{change_col}_tukey"] = False
 if not (np.isnan(low) or np.isnan(high)):
 out[f"flag_{change_col}_tukey"] = (out[change_col] < low) | (out[change_col] > high)
 out[f"{change_col}_low"] = low
 out[f"{change_col}_high"] = high
 return out

def seasonal_factors_by_group(df_group, group_col):

        .agg(mean_revenue=("revenue","mean"),
             group_mean=("group_mean","first"))
    )
    
    seas["seasonal_factor"] = seas["mean_revenue"] / seas["group_mean"]
