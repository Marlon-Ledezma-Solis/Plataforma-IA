import os, time

class PdfHandler(FileSystemEventHandler):
    def on_created(self, event):
        if event.is_directory:
            return
        if not event.src_path.lower().endswith(".pdf"):
            return
        
        # Esperar a que termine la copia
        time.sleep(1.0)
        
        pdf_path = event.src_path
        print("[+] PDF nuevo:", os.path.basename(pdf_path))
        
        # Pipeline completo:
        # 1) Convertir/log/email
        # 2) IA -> embedding -> PCA -> score
Patrón de Automatización
El FileSystemEventHandler detecta eventos de creación de archivos. Implementamos un delay de 1 segundo para asegurar que el archivo esté completamente copiado antes de procesarlo.
Este patrón es productivo porque permite procesamiento en tiempo real sin polling constante, reduciendo uso de recursos.

from pdf2docx import Converter

def pdf_to_docx(pdf_path, docx_path):
    cv = Converter(pdf_path)
    try:
        cv.convert(docx_path, start=0, end=None)
    finally:
        cv.close()
Función simple pero robusta que maneja la conversión completa del documento. El bloque try-finally asegura que el conversor se cierre correctamente incluso si hay errores.

Representación Conceptual
El embedding NO es "bolsa de palabras"; es una representación conceptual del texto en espacio vectorial de 256 dimensiones.
text-embedding-3-large es el modelo más capaz de OpenAI y permite reducir dimensiones sin perder propiedades conceptuales mediante el parámetro dimensions.

--- Diapositiva 16 ---
Implementación de Embeddings
import numpy as np

EMB_MODEL = "text-embedding-3-large"
EMB_DIM = 256  # Vector compacto para eficiencia

def embed_semantic(text: str) -> np.ndarray:
    emb = client.embeddings.create(
        model=EMB_MODEL,
        input=text,
        dimensions=EMB_DIM
    )
    return np.array(emb.data[0].embedding, dtype=np.float32)
Usamos float32 en lugar de float64 para reducir el tamaño del BLOB en la base de datos sin pérdida significativa de precisión.
Este embedding captura el significado semántico del resumen de riesgo, permitiendo comparaciones conceptuales entre solicitudes.

import sqlite3
from sklearn.decomposition import PCA
from joblib import dump, load

PCA_PATH = "./pca_model.joblib"
N_PCA = 8

def ensure_pca_model(current_embedding: np.ndarray, 
 min_samples_to_fit: int = 15):
 # Si existe modelo, usarlo
 if os.path.exists(PCA_PATH):
 return load(PCA_PATH)
 
 # Si no existe, intentar entrenar con históricos
 conn = sqlite3.connect(DB_PATH)
 cur = conn.cursor()
 cur.execute("""SELECT embedding FROM credit_ai_log 
 WHERE embedding IS NOT NULL AND status='ok'""")
 rows = cur.fetchall()
 conn.close()
 
 vecs = []
 for (blob,) in rows:
 v = np.frombuffer(blob, dtype=np.float32)
 if v.shape[0] == EMB_DIM:
 vecs.append(v)

--- Diapositiva 19 ---
Entrenamiento Incremental de PCA
# Incluir el embedding actual
    vecs.append(current_embedding)
    
    if len(vecs) < min_samples_to_fit:
        return None  # Aún no hay suficientes muestras
    
    # Entrenar PCA con todos los embeddings disponibles
    X = np.vstack(vecs)
    pca = PCA(n_components=N_PCA, random_state=42).fit(X)
    
    # Guardar modelo para uso futuro
    dump(pca, PCA_PATH)
    
    return pca
Este enfoque permite que el sistema comience a funcionar inmediatamente y mejore automáticamente a medida que procesa más documentos.

def process_pdf_with_ai(pdf_path: str):
 pdf_filename = os.path.basename(pdf_path)
 
 # A) IA analiza (sin score todavía)
 analysis = llm_analyze_pdf(pdf_path)
 summary_text = analysis["risk_summary"]["summary_text"]
 
 # B) Embedding semántico del resumen
 emb_vec = embed_semantic(summary_text)
 
 # C) Intentar obtener/entrenar modelo PCA
 pca = ensure_pca_model(emb_vec, min_samples_to_fit=15)
 
 if pca is None:
 # Guardar análisis + embedding para entrenar PCA después
 db_insert_ai_log(pdf_filename, analysis, summary_text, 
 emb_vec, None, None, None, "ok",
 "PCA aún no entrenado (faltan muestras).")
 return {"status":"ok_no_pca_yet", "pdf": pdf_filename}

--- Diapositiva 32 ---
Función Main y Ejecución
def main():
    os.makedirs(INBOX_DIR, exist_ok=True)
    os.makedirs(OUT_DIR, exist_ok=True)
    os.makedirs(PROCESSED_DIR, exist_ok=True)
    
    init_db()
    
    observer = Observer()
    observer.schedule(PdfHandler(), INBOX_DIR, recursive=False)
    observer.start()
    
    print(f"Escuchando {INBOX_DIR} ... CTRL+C para salir")
    
    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        observer.stop()
    observer.join()

if __name__ == "__main__":
    main()

--- Diapositiva 38 ---
ROL DE LA IA
IA como "Minero de Insights"
Qué Hace Bien la IA
Interpretar resultados (qué significa)
Priorizar señales (qué importa más)
Generar hipótesis (posibles causas)
Proponer acciones (qué hacer después)
Qué NO Hace
La IA no reemplaza el cálculo estadístico. Los análisis RFM, outliers, MoM/YoY y estacionalidad deben calcularse con código determinístico.
La IA interpreta, no calcula.

--- Diapositiva 49 ---
Función de Minería con Artefactos
def ai_mine_insights_from_artifacts(artifacts: dict):
    resp = client.responses.create(
        model="gpt-4o-mini",
        input=[
            {"role":"system","content": SYSTEM_ANALYST},
            {"role":"user","content":
                "Analiza estos artefactos y entrega hallazgos "
                "profundos y accionables.\n"
                + json.dumps(artifacts, ensure_ascii=False)
            }
        ],
        response_format={"type":"json_schema", "json_schema": INSIGHT_SCHEMA}
    )
    
    return json.loads(resp.output_text)
Esta función envía el paquete compacto de artefactos y recibe hallazgos estructurados con evidencia, hipótesis y acciones recomendadas.

            }
        ]
    )
    
    return resp.output_text

Capa 4: Trazabilidad y Reproducibilidad
import json
from datetime import datetime

def save_analysis_run(artifacts, findings_json, narrative, version="1.0"):
    timestamp = datetime.utcnow().isoformat()
    
    run_data = {
        "timestamp": timestamp,
        "pipeline_version": version,
        "artifacts": artifacts,
        "findings": findings_json,
        "narrative": narrative,
        "system_prompt": SYSTEM_ANALYST,
        "schema": INSIGHT_SCHEMA
    }
    
    filename = f"analysis_run_{timestamp.replace(':','-')}.json"
    with open(filename, 'w', encoding='utf-8') as f:
        json.dump(run_data, f, ensure_ascii=False, indent=2)
    
    print(f"[+] Análisis guardado: {filename}")
    return filename
Guardar todo permite reproducir exactamente el análisis y auditar decisiones posteriores.
